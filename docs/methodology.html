Step 2.1: Download 311 Streetlight Data

Code 
import pandas as pd
import geopandas as gpd
from datetime import datetime

# Download from OpenDataPhilly
url_311 = "https://phl.carto.com/api/v2/sql?q=SELECT * FROM public_cases_fc WHERE service_name = 'Streetlight Outage'&format=csv"

# Read data
df_311 = pd.read_csv(url_311)

# Convert to datetime
df_311['requested_datetime'] = pd.to_datetime(df_311['requested_datetime'])
df_311['closed_date'] = pd.to_datetime(df_311['closed_date'])

# Calculate repair delay (in days)
df_311['repair_delay'] = (df_311['closed_date'] - df_311['requested_datetime']).dt.days

# Convert to GeoDataFrame
gdf_311 = gpd.GeoDataFrame(
    df_311,
    geometry=gpd.points_from_xy(df_311.lon, df_311.lat),
    crs='EPSG:4326'
)

# Save
gdf_311.to_file('data/processed/streetlight_outages.geojson', driver='GeoJSON')
print(f"✓ Downloaded {len(gdf_311)} streetlight outage records")

Phase 3: Spatial Unit Creation

Create Analysis Grid (Hexagons)

Code 
import geopandas as gpd
from shapely.geometry import box
import h3

# Load Philadelphia boundary
philly = gpd.read_file('data/raw/philly_boundary.geojson')

# Create H3 hexagon grid (resolution 8 ≈ 0.46 km² per cell)
bounds = philly.total_bounds
minx, miny, maxx, maxy = bounds

# Generate points across the area
import numpy as np
lats = np.linspace(miny, maxy, 100)
lons = np.linspace(minx, maxx, 100)

hexagons = set()
for lat in lats:
    for lon in lons:
        h = h3.geo_to_h3(lat, lon, 8)
        hexagons.add(h)

# Convert to GeoDataFrame
from shapely.geometry import Polygon

def h3_to_polygon(h):
    coords = h3.h3_to_geo_boundary(h, geo_json=True)
    return Polygon(coords)

gdf_hex = gpd.GeoDataFrame(
    {'hex_id': list(hexagons)},
    geometry=[h3_to_polygon(h) for h in hexagons],
    crs='EPSG:4326'
)

# Clip to Philadelphia boundary
gdf_hex = gpd.overlay(gdf_hex, philly, how='intersection')

gdf_hex.to_file('data/spatial/analysis_hexagons.geojson', driver='GeoJSON')
print(f"✓ Created {len(gdf_hex)} hexagonal analysis units")

Phase 4: Compute Lighting Metrics

Calculate Outage Density (OD) per Hexagon

Code 
# Load data
gdf_311 = gpd.read_file('data/processed/streetlight_outages.geojson')
gdf_hex = gpd.read_file('data/spatial/analysis_hexagons.geojson')

# Spatial join: count outages per hexagon
gdf_hex = gdf_hex.to_crs(gdf_311.crs)
joined = gpd.sjoin(gdf_311, gdf_hex, how='inner', predicate='within')

# Aggregate
outage_counts = joined.groupby('hex_id').size().reset_index(name='outage_count')
gdf_hex = gdf_hex.merge(outage_counts, on='hex_id', how='left')
gdf_hex['outage_count'] = gdf_hex['outage_count'].fillna(0)

# Calculate density (outages per km²)
gdf_hex['area_km2'] = gdf_hex.to_crs('EPSG:3857').area / 1e6
gdf_hex['outage_density'] = gdf_hex['outage_count'] / gdf_hex['area_km2']

print("✓ Outage Density calculated")

Calculate Mean Repair Delay (RD)

Code 
# Calculate mean repair delay per hexagon
repair_delays = joined.groupby('hex_id')['repair_delay'].mean().reset_index(name='mean_repair_delay')
gdf_hex = gdf_hex.merge(repair_delays, on='hex_id', how='left')
gdf_hex['mean_repair_delay'] = gdf_hex['mean_repair_delay'].fillna(0)

print("✓ Repair Delay calculated")

Phase 5: Lighting Equity Index (LEI)

Integrate Socioeconomic Data

Code 
# Load census data
df_acs = pd.read_csv('data/processed/acs_census_data.csv')

# Load census tract boundaries
gdf_tracts = gpd.read_file('https://www2.census.gov/geo/tiger/TIGER2022/TRACT/tl_2022_42_tract.zip')
gdf_tracts = gdf_tracts[gdf_tracts['COUNTYFP'] == '101']  # Philadelphia

# Merge ACS data with tract geometries
gdf_tracts['GEOID'] = gdf_tracts['GEOID'].astype(str)
df_acs['tract'] = df_acs['state'] + df_acs['county'] + df_acs['tract']
gdf_tracts = gdf_tracts.merge(df_acs, left_on='GEOID', right_on='tract', how='left')

# Spatial join with hexagons (area-weighted average)
from tobler.area_weighted import area_interpolate

gdf_hex_socio = area_interpolate(
    source_df=gdf_tracts,
    target_df=gdf_hex,
    extensive_variables=['poverty_count', 'no_vehicle_households'],
    intensive_variables=['median_income']
)

print("✓ Socioeconomic data integrated")

Calculate LEI Score

Code 
from sklearn.preprocessing import StandardScaler

# Select indicators
indicators = ['outage_density', 'mean_repair_delay', 'poverty_count', 'no_vehicle_households']
X = gdf_hex_socio[indicators].fillna(0)

# Standardize
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# Simple weighted average (equal weights)
weights = [0.3, 0.3, 0.2, 0.2]
gdf_hex_socio['LEI'] = (X_scaled * weights).sum(axis=1)

# Normalize to 0-100 scale
gdf_hex_socio['LEI'] = ((gdf_hex_socio['LEI'] - gdf_hex_socio['LEI'].min()) / 
                         (gdf_hex_socio['LEI'].max() - gdf_hex_socio['LEI'].min())) * 100

gdf_hex_socio.to_file('data/spatial/hexagons_with_LEI.geojson', driver='GeoJSON')
print("✓ Lighting Equity Index calculated")

Phase 6: Spatial Autocorrelation

Moran’s I for LEI

Code 
from libpysal.weights import Queen
from esda.moran import Moran
import matplotlib.pyplot as plt

# Create spatial weights
w = Queen.from_dataframe(gdf_hex_socio)
w.transform = 'r'

# Calculate Moran's I
moran = Moran(gdf_hex_socio['LEI'], w)

print(f"Moran's I: {moran.I:.3f}")
print(f"P-value: {moran.p_sim:.4f}")
print(f"Interpretation: {'Clustered' if moran.p_sim < 0.05 else 'Random'}")

# Plot
fig, ax = plt.subplots(figsize=(8, 6))
from splot.esda import plot_moran
plot_moran(moran, ax=ax)
plt.savefig('outputs/figures/morans_i_lei.png', dpi=300, bbox_inches='tight')

LISA Cluster Map

Code 
from esda.moran import Moran_Local
import matplotlib.pyplot as plt

# Local Moran's I
lisa = Moran_Local(gdf_hex_socio['LEI'], w)

# Add LISA categories to GeoDataFrame
gdf_hex_socio['lisa_cluster'] = lisa.q

# Map categories: 1=HH, 2=LH, 3=LL, 4=HL
cluster_labels = {1: 'High-High', 2: 'Low-High', 3: 'Low-Low', 4: 'High-Low', 0: 'Not Significant'}
gdf_hex_socio['cluster_label'] = gdf_hex_socio['lisa_cluster'].map(cluster_labels)

# Plot
fig, ax = plt.subplots(figsize=(12, 10))
gdf_hex_socio.plot(column='cluster_label', ax=ax, legend=True, categorical=True, cmap='RdYlBu')

Phase 7: Crime-Lighting Regression

Spatial Lag Model

Code 
from spreg import OLS, ML_Lag
import numpy as np

# Aggregate crime counts per hexagon
gdf_crime = gpd.read_file('data/processed/crime_nighttime.geojson')
crime_counts = gpd.sjoin(gdf_crime, gdf_hex_socio, how='inner').groupby('hex_id').size()
gdf_hex_socio['crime_count'] = gdf_hex_socio['hex_id'].map(crime_counts).fillna(0)

# Crime rate per km²
gdf_hex_socio['crime_rate'] = gdf_hex_socio['crime_count'] / gdf_hex_socio['area_km2']

# Prepare variables
y = gdf_hex_socio['crime_rate'].values.reshape(-1, 1)
X = gdf_hex_socio[['outage_density', 'mean_repair_delay', 'median_income', 'poverty_count']].fillna(0).values

# Spatial Lag Model
model = ML_Lag(y, X, w=w, name_y='crime_rate', 
               name_x=['outage_density', 'repair_delay', 'income', 'poverty'])

print(model.summary)

Phase 8: Night Transit Exposure (NTE)

Compute Transit Accessibility

Code 
# Load SEPTA stops
gdf_stops = gpd.read_file('data/processed/septa_stops.geojson')

# Create 400m (10-min walk) buffers
gdf_stops_buffered = gdf_stops.to_crs('EPSG:3857')
gdf_stops_buffered['geometry'] = gdf_stops_buffered.buffer(400)

# Spatial join with hexagons
gdf_hex_transit = gpd.sjoin(gdf_hex_socio, gdf_stops_buffered.to_crs(gdf_hex_socio.crs), 
                              how='left', predicate='intersects')

# Count stops within buffer
transit_access = gdf_hex_transit.groupby('hex_id').size().reset_index(name='stops_nearby')
gdf_hex_socio = gdf_hex_socio.merge(transit_access, on='hex_id', how='left')
gdf_hex_socio['stops_nearby'] = gdf_hex_socio['stops_nearby'].fillna(0)

# Calculate NTE
gdf_hex_socio['NTE'] = (gdf_hex_socio['stops_nearby'] * gdf_hex_socio['LEI']) / 100

gdf_hex_socio.to_file('data/spatial/hexagons_final.geojson', driver='GeoJSON')
print("✓ Night Transit Exposure calculated")

Phase 9: Predictive Modeling

Random Forest for Dark Zone Prediction

Code 
from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import train_test_split
from sklearn.metrics import classification_report, roc_auc_score

# Define "dark zone" threshold (e.g., LEI > 75th percentile)
threshold = gdf_hex_socio['LEI'].quantile(0.75)
gdf_hex_socio['is_dark_zone'] = (gdf_hex_socio['LEI'] > threshold).astype(int)

# Features
features = ['outage_density', 'mean_repair_delay', 'poverty_count', 
            'no_vehicle_households', 'stops_nearby', 'crime_rate']
X = gdf_hex_socio[features].fillna(0)
y = gdf_hex_socio['is_dark_zone']

# Split data
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

# Train Random Forest
rf = RandomForestClassifier(n_estimators=100, random_state=42)
rf.fit(X_train, y_train)

# Evaluate
y_pred = rf.predict(X_test)
print(classification_report(y_test, y_pred))
print(f"ROC-AUC: {roc_auc_score(y_test, rf.predict_proba(X_test)[:, 1]):.3f}")

# Feature importance
import pandas as pd
importance = pd.DataFrame({
    'feature': features,
    'importance': rf.feature_importances_
}).sort_values('importance', ascending=False)

print("\nFeature Importance:")
print(importance)

# Phase 3: Spatial Unit Creation {#sec-spatial-units}

## Create Analysis Grid (Hexagons)

```{python}
import geopandas as gpd
from shapely.geometry import box
import h3

# Load Philadelphia boundary
philly = gpd.read_file('data/raw/philly_boundary.geojson')

# Create H3 hexagon grid (resolution 8 ≈ 0.46 km² per cell)
bounds = philly.total_bounds
minx, miny, maxx, maxy = bounds

# Generate points across the area
import numpy as np
lats = np.linspace(miny, maxy, 100)
lons = np.linspace(minx, maxx, 100)

hexagons = set()
for lat in lats:
    for lon in lons:
        h = h3.geo_to_h3(lat, lon, 8)
        hexagons.add(h)

# Convert to GeoDataFrame
from shapely.geometry import Polygon

def h3_to_polygon(h):
    coords = h3.h3_to_geo_boundary(h, geo_json=True)
    return Polygon(coords)

gdf_hex = gpd.GeoDataFrame(
    {'hex_id': list(hexagons)},
    geometry=[h3_to_polygon(h) for h in hexagons],
    crs='EPSG:4326'
)

# Clip to Philadelphia boundary
gdf_hex = gpd.overlay(gdf_hex, philly, how='intersection')

gdf_hex.to_file('data/spatial/analysis_hexagons.geojson', driver='GeoJSON')
print(f"✓ Created {len(gdf_hex)} hexagonal analysis units")
```

```{python}
# Load data
gdf_311 = gpd.read_file('data/processed/streetlight_outages.geojson')

gdf_hex = gdf_hex.to_crs(gdf_311.crs)
joined = gpd.sjoin(gdf_311, gdf_hex, how='inner', predicate='within')

outage_counts = joined.groupby('hex_id').size().reset_index(name='outage_count')

gdf_hex['area_km2'] = gdf_hex.to_crs('EPSG:3857').area / 1e6
gdf_hex['outage_density'] = gdf_hex['outage_count'] / gdf_hex['area_km2']
```

```{python}
    intensive_variables=['median_income']
```

## Calculate LEI Score

```{python}
from sklearn.preprocessing import StandardScaler

# Select indicators
indicators = ['outage_density', 'mean_repair_delay', 'poverty_count', 'no_vehicle_households']
X = gdf_hex_socio[indicators].fillna(0)

# Standardize
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# Simple weighted average (equal weights)
weights = [0.3, 0.3, 0.2, 0.2]
gdf_hex_socio['LEI'] = (X_scaled * weights).sum(axis=1)

# Normalize to 0-100 scale
gdf_hex_socio['LEI'] = ((gdf_hex_socio['LEI'] - gdf_hex_socio['LEI'].min()) / 
                         (gdf_hex_socio['LEI'].max() - gdf_hex_socio['LEI'].min())) * 100

gdf_hex_socio.to_file('data/spatial/hexagons_with_LEI.geojson', driver='GeoJSON')
```

```{python}
from sklearn.ensemble import RandomForestClassifier
```
